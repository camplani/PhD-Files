----------------------------------------------------------------------------------------------------
-- Put a command in background:
----------------------------------------------------------------------------------------------------
nohup ./write_counter x.y.z.a b 0</dev/null &

----------------------------------------------------------------------------------------------------
-- Find out if write_counter is running:
-- Try this after you logged out and logged in again to the machine 
-- If still there, it's really running
----------------------------------------------------------------------------------------------------
ps -a | grep write

----------------------------------------------------------------------------------------------------
-- How to kill a command:
-- PID is the number that you get from the ps command
----------------------------------------------------------------------------------------------------
kill <PID>

----------------------------------------------------------------------------------------------------
-- Check for words in a file: 
-- The first grep checks in nohup.out for the timeout message 
-- The second grep only selects those with a given (part of a) timestamp
-- Head -n 1 only prints out the first result
----------------------------------------------------------------------------------------------------
grep "Timeout" nohup.out | grep "<time>" | head -n 1

----------------------------------------------------------------------------------------------------
-- how to count the line number in a file
----------------------------------------------------------------------------------------------------
wc -l <filename>

----------------------------------------------------------------------------------------------------
-- how to arrive to see ABBA readout data/abbaReadout
----------------------------------------------------------------------------------------------------
/data/abbaReadout

----------------------------------------------------------------------------------------------------
-- list the last 10 file recorded
----------------------------------------------------------------------------------------------------
ls -hlrt | tail -n 10
